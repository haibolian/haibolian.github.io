import VueRouter from 'vue-router' //引号内可以写具体的路径名称，也可直接写模块名称

​ import 组件名称 from “ 组件路径 ”

# 1.3 创建路由对象

var router = new Vuerouter({
	routers:[  //配置路由匹配规则
		{ path: ' 地址 ' , component: 组件名称 },  
		{ path: '/home/newsinfo/:id', component: NewsInfo }, 
		{ path: '/home/goodsinfo/:id', component: GoodsInfo, name: 'goodsinfo' },
	]
})

​ component：组成部分，部件，成分。

​ 普通路由规则由为path和component组成，path中填写路由地址的名称，component为组件的名称。

​ 第二条路由规则有个 :id ，向URL地址传递参数。

<router-link :to="'/home/newsinfo/' + item.id">

​ 上面的代码即跳转到指定id参数的页面，绑定to，地址用字符串拼接起来

​ 使用this.$route.params.id 则获取地址中传递过来的值

​ 第三条路由规则涉及到编程式导航，为组件声明name，如需跳转到该模板页面，点击触发函数，在函数内声明如下代码，params中是向URL地址传递的参数

this.$router.push({ name: "goodsinfo", params: { id } });

# 3. App.vue组件知识点总结

# 3.1 返回按键

v-show="flag"，通过控制flag的布尔值，从而控制返回按钮是否在首页进行展示，点击按钮，可触发返回函数

 this.$router.go(-1);		//或者为  this.$router.back()

通过观察判断路径是否为首页，使flag是否为真，因此要创建一个动态的判断flag的代码

watch: {
    "$route.path": function(newVal) {
      if (newVal === "/home") {
        this.flag = false;
      } else {
        this.flag = true;
      }}}

上面的思路看起来没有问题，但当进入不是首页的任意一个页面时，点击刷新，此时返回键会消失，这是因为刷新的时候路径地址并没有发生变化，watch中的内容不执行，而flag默认值为false，所以刷新时返回键会消失；如果将flag默认值改为true，那么在首页的时候，刷新后返回键会出现，因此不论flag默认值是否为true还是false，刷新页面时都会有bug，所以要在钩子函数created中写判断路径来设置flag值的代码

this.flag= this.$route.path==="/home"? false:true

# 3.2 页面的进场离场动画

  1. 将需要设置动画的标签或者组件用transition标签包裹起来

  2. 在css中设置动画，.v-enter可以为需要动画的内容设置开始进场第一时刻，如设置opacity透明度，transform:translate(100%)；也可以针对X或Y来设置；v-leave-to是离开的最后一刻，一般和v-enter一起使用，但此动画的效果是从右边进入，左边离开，因此在v-leave-to中，使transform中的translateX设置为-100%，而且要设置绝对定位，如果不设置绝对定位的话，第一个页面要完全离开后，第二个页面才会进入，或者第二个页面在下面的位置。因此让要离开的页面脱离标准流，动画效果才会正常；设置动画效果：

    .v-enter-active,.v-leave-active {
  transition: all 0.5s ease;
}

# 7.3 tag

有时使用了router-link后，router-link在渲染的时候，会默认渲染成a标签，但如果之前在做样式的时候使用的是div这种和a标签有很大不同的标签时，样式会产生变化，所以要想使router-link渲染时变成自己想要的标签，则在router-link标签的属性中添加： tag="想要变成的标签名" （区分tap和tag）
