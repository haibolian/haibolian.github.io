# 浅拷贝与深拷贝

# 1. 浅拷贝

# 1.1 object.assign

function cloneshallow(target){
  let cloneTarget = {}
  return Object.assign(cloneTarget,target)
}

let target = {
  name:'ming',
  age:18,
  hobby:['篮球','足球'],
  companies:{
    one:'中国建设银行',
    two:'中国工商银行'
  }
}

let cloneTarget = cloneshallow(target)
console.log('cloneTarget',cloneTarget); // false
cloneTarget.companies.one = '招商银行'
console.log(target.companies); // {one: "招商银行", two: "中国工商银行"}

从上面可以看到 object.assign 可以实现浅拷贝，但它有几点要注意：

  • 它不会拷贝对象的继承属性；
  • 它不会拷贝对象的不可枚举的属性；
  • 可以拷贝 Symbol 类型的属性。
// 它不会拷贝对象的继承属性
class Person {}
let obj = { a:{ b:1 } , b : 2 }
obj.__proto__ = Person
console.log(obj);// obj.__proto__ = Person
console.log(Object.assign({},obj)); // {}.__proto__ = Object

// 它不会拷贝对象的不可枚举的属性
let obj = { a:{ b:1 } , b : 2 , sym : Symbol(1)}
Object.defineProperty(obj,'b',{
	enumerable:false
})
// 可以拷贝 Symbol 类型的属性
console.log(Object.assign({},obj)); // { a:{b:1}, sym:Symbol(1) }

# 1.2 扩展运算符

class Person {}
let obj = { a:{ b:1 } , b : 2 ,sym:Symbol(12)}
obj.__proto__ = Person
console.log(obj);
Object.defineProperty(obj,'b',{
	enumerable:false
})
console.log({ ...obj });

扩展运算符的方式和 object.assign 方式运行的结果相同，同样有以下几点需要注意：

  • 它不会拷贝对象的继承属性；
  • 它不会拷贝对象的不可枚举的属性；
  • 可以拷贝 Symbol 类型的属性。

# 1.3 数组concat

concat 也是一个浅拷贝。因此，当用 concat 链接两个数组，后期修改原数组时，需要考虑其中的引用类型，因为它会影响该连接后的的数组。

let arr = [1, 2, 3];
let newArr = arr.concat();
newArr[1] = 100;
console.log(arr);  // [ 1, 2, 3 ]
console.log(newArr); // [ 1, 100, 3 ]

# 1.4 数组slice

和 concat 一样，slice 只是拷贝第一层

let arr = [1, 2, {val: 4}];
let newArr = arr.slice();
newArr[2].val = 1000;
console.log(arr);  //[ 1, 2, { val: 1000 } ]
